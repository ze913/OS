class 电梯:
    def __init__(编号):
        楼层 = 1
        状态 = 停止
        队列 = 空队列

    def run():
        while 一直循环:
            if 队列空:
                休息
            else:
                目标 = 取队首
                if 当前 < 目标: 上升
                elif 当前 > 目标: 下降
                else: 开门

    def add任务(楼层):
        加入队列


class 主窗口:
    def __init__():
        创建 5 个电梯

    def dispatch(目标楼层):
        遍历 5 个电梯:
            找到最合适的
        给它加任务


按钮点击 → 调用 dispatch
